argus-ops has been deployed.

1. Get the application URL:
{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}/
{{- end }}
{{- else if contains "NodePort" .Values.service.type }}
  export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
  export NODE_PORT=$(kubectl get svc {{ include "argus-ops.fullname" . }} -n {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}")
  echo "Dashboard: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
  export SERVICE_IP=$(kubectl get svc {{ include "argus-ops.fullname" . }} -n {{ .Release.Namespace }} --template "{{"{{range (index .status.loadBalancer.ingress 0)}}{{.}}{{end}}"}}")
  echo "Dashboard: http://$SERVICE_IP:{{ .Values.service.port }}"
{{- else }}
  kubectl port-forward svc/{{ include "argus-ops.fullname" . }} 8080:{{ .Values.service.port }} -n {{ .Release.Namespace }}
  echo "Dashboard: http://localhost:8080"
{{- end }}

2. Create the AI provider secret (if not already done):
   kubectl create secret generic argus-ops-secrets \
     --from-literal={{ .Values.existingSecret.key }}=<your-api-key> \
     -n {{ .Release.Namespace }}

   Then set existingSecret.name in your values:
     helm upgrade {{ .Release.Name }} argus-ops/argus-ops \
       --set existingSecret.name=argus-ops-secrets

3. Check pod status:
   kubectl get pods -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "argus-ops.name" . }}

4. View logs:
   kubectl logs -n {{ .Release.Namespace }} -l app.kubernetes.io/name={{ include "argus-ops.name" . }} --tail=50
